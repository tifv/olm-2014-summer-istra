\section*{Диофантовы уравнения}

% $authors:
% - Юлий Тихонов
% - Михаил Ягудин

\emph{%
Решите уравнения, если не предлагается иное.
Решите в целых числах, если не указано иное.}

\subsection*{Линейные уравнения}

\begin{problems}

\item
$3n + 15 m = 7$;
\qquad
$5 x - 7 y = 3$;
\qquad
$55 x - 41 y = 444$.

\end{problems}

\subsection*{Метод разложения на множители}

\begin{problems}

\item
$x^2 - y^2 = 2011$\; в $\NN$.
\qquad
\problem
$x^3 + x^2 + x + 3 = 0$.
\qquad
\problem
$x^2 + x = y^2$.

\end{problems}

\subsection*{Метод остатков}

\begin{problems}

\item
Докажите, что следующие уравнения не имеют решений в $\ZZ$:
\\
\sbp $x^2 + y^2 = 2011$;
%^2\mod4 
\qquad
\sbp $a^2 - 3 b^2 = 8$;
%^2\mod3 
\qquad
\sbp $x^2 + y^2 + z^2 = 7^{2011}$.
%^2\mod8 

\item
$3^m + 7 = 2^n$.
%m=4
\qquad
\problem
$1! + 2! + \ldots + n! = m^2$\; в $\NN$.
%m=8

\end{problems}

\qquad
